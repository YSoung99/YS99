    <script>

        const sourceTextarea = document.getElementById('source-text');

        const translationResultArea = document.getElementById('translation-result');

        const translateButton = document.getElementById('translate-button');

        const clearButton = document.getElementById('clear-button');

        const errorMessage = document.getElementById('error-message');

        const apiKeyInput = document.getElementById('api-key-input');

        const modelSelect = document.getElementById('model-select');


        // Gemini API 키를 여기에 입력하세요.

        let API_KEY = '';

        let API_URL = 'https://generativelanguage.googleapis.com/v1beta/models/';


        function showError(message) {

            errorMessage.textContent = message;

            translationResultArea.value = '';

        }


        function clearInputAndResult() {

            sourceTextarea.value = '';

            translationResultArea.value = '';

            errorMessage.textContent = '';

            apiKeyInput.value = '';

            API_KEY = '';

        }


        async function translateText() {

            const sourceText = sourceTextarea.value.trim();

            API_KEY = apiKeyInput.value.trim();

            const selectedModel = modelSelect.value;


            if (!sourceText) {

                showError('번역할 텍스트를 입력해주세요.');

                return;

            }


            if (!API_KEY) {

                showError('API 키를 입력해주세요.');

                return;

            }


            errorMessage.textContent = '';

            translationResultArea.value = '번역 중...';


            try {

                // 선택된 모델에 따라 API URL을 설정합니다.

                const fullApiUrl = `${API_URL}${selectedModel}:generateContent?key=${API_KEY}`;

                console.log("API URL:", fullApiUrl);


                const response = await fetch(fullApiUrl, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json',

                    },

                    body: JSON.stringify({

                        contents: [{

                            parts: [{

                                text: `Translate the following text to Korean in a professional business style.  Just return the translated text: ${sourceText}`

                            }]

                        }]

                    }),

                });


                if (!response.ok) {

                    const errorText = await response.text();

                    console.error(`HTTP error! status: ${response.status}, response text: ${errorText}`);

                    throw new Error(`HTTP error! status: ${response.status}, ${errorText}`);

                }


                const data = await response.json();

                console.log(data);


                if (data.candidates && data.candidates[0]?.content?.parts && data.candidates[0].content.parts[0]?.text) {

                    const translatedText = data.candidates[0].content.parts[0].text;

                    translationResultArea.value = translatedText;

                } else {

                    throw new Error('번역 결과를 받지 못했습니다.');

                }


            } catch (error) {

                console.error('Error:', error);

                showError(`번역 중 오류가 발생했습니다: ${error.message}`);

            }

        }


        translateButton.addEventListener('click', translateText);

        clearButton.addEventListener('click', clearInputAndResult);

    </script>
